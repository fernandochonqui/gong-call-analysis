-- Gong calls for closed-won accounts (last 12 months)
-- Tables: CLEAN.GONG.CALLS__WRK, CLEAN.GONG.CONVERSATION_CONTEXTS,
-- CLEAN.SALESFORCE.SALESFORCE__OPPORTUNITY__WRK, CLEAN.SALESFORCE.ACCOUNT

WITH closed_won_opps AS (
  SELECT DISTINCT
    o.ACCOUNTID,
    a.NAME AS account_name,
    a.COMPANY_SIZE__C,
    a.SUBSCRIPTION_PLAN_NAME__C AS plan,
    o.ARR__C,
    o.TYPE AS deal_type,
    o.CLOSEDATE
  FROM CLEAN.SALESFORCE.SALESFORCE__OPPORTUNITY__WRK o
  JOIN CLEAN.SALESFORCE.ACCOUNT a ON a.ID = o.ACCOUNTID
  WHERE o.STAGENAME = '(5) Closed Won'
    AND o.ISDELETED = FALSE
    AND o.CLOSEDATE >= DATEADD('month', -12, CURRENT_DATE())
    AND o.TYPE IN ('New Business', 'Upsell', 'Expansion', 'Renewal')
),
gong_calls AS (
  SELECT
    cc.object_id AS salesforce_account_id,
    c.conversation_key,
    c.title,
    c.planned_start_datetime,
    c.duration_seconds,
    c.CALL_SPOTLIGHT,
    -- Classify call type from title
    CASE
      WHEN LOWER(c.title) LIKE '%discovery%' THEN 'Discovery'
      WHEN LOWER(c.title) LIKE ANY ('%demo%', '%walkthrough%', '%overview%') THEN 'Demo'
      WHEN LOWER(c.title) LIKE ANY ('%technical%', '%poc%', '%proof of concept%', '%api%') THEN 'Technical / POC'
      WHEN LOWER(c.title) LIKE ANY ('%proposal%', '%pricing%', '%commercial%') THEN 'Proposal Review'
      WHEN LOWER(c.title) LIKE ANY ('%negotiat%', '%contract review%', '%renewal%') THEN 'Negotiation'
      WHEN LOWER(c.title) LIKE ANY ('%qbr%', '%account sync%', '%check-in%', '%check in%') THEN 'QBR / Account Sync'
      ELSE 'Other'
    END AS call_type
  FROM CLEAN.GONG.CALLS__WRK c
  JOIN CLEAN.GONG.CONVERSATION_CONTEXTS cc
    ON cc.conversation_key = c.conversation_key
    AND cc.object_type = 'account'
  WHERE c.status = 'COMPLETED'
    AND c.CALL_SPOTLIGHT IS NOT NULL
    AND c.planned_start_datetime >= DATEADD('month', -18, CURRENT_DATE())
)
SELECT
  o.account_name,
  o.deal_type,
  o.company_size,
  o.plan,
  o.arr,
  g.call_type,
  COUNT(*) AS call_count,
  MIN(g.planned_start_datetime) AS first_call,
  MAX(g.planned_start_datetime) AS last_call,
  DATEDIFF('day', MIN(g.planned_start_datetime), o.closedate) AS days_from_first_call_to_close
FROM closed_won_opps o
JOIN gong_calls g ON g.salesforce_account_id = o.accountid
GROUP BY 1,2,3,4,5,6
ORDER BY o.account_name, g.call_type
LIMIT 500;
